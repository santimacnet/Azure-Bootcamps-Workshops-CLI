PRACTICA PARA MEETUP BCN
--------------------------------------------------------
$ az login
$ az group create --name MeetupBcn --location northeurope

# Configure ACR y WebApp
$ az acr create -n meetupreg -g MeetupBcn-recursos --sku Standard --admin-enabled true
$ az appservice plan create -n MeetupPlanPrice -g MeetupBcn-recursos --is-linux
$ az webapp create -n MeetupWebapp -g MeetupBcn-recursos -p MeetupPlanPrice -i elnably/dockerimagetest

# Configure SQL Server
$ az sql server create -l northeurope -g MeetupBcn-recursos -n meetupsql -u sqladmin -p Password123
$ az sql db create -g MeetupBcn-recursos -s MeetupSql -n meetupclinicdb --service-objective S0
$ az sql server firewall-rule create --resource-group MeetupBcn-recursos --server meetupsql --name AllowAllAzureIps --start-ip-address 0.0.0.0 --end-ip-address 0.0.0.0

# Configure ConnectionString SQL
$ az webapp config connection-string set -g MeetupBcn-recursos -n MeetupWebapp -t SQLAzure --settings defaultConnection='Data Source=tcp:meetupsql.database.windows.net,1433;Initial Catalog=meetupclinicdb;User Id=sqladmin;Password=Password123;'
